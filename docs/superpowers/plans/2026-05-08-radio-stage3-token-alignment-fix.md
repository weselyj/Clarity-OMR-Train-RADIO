# RADIO Stage 3 Token Alignment Audit + Fix

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate silent token/image misalignment in `synthetic_systems_v1` (and confirm the other Stage 3 corpora are clean) by fixing the per-staff manifest writer to emit pre-filter physical `staff_index` values, regenerating downstream artifacts, and locking the contract with tests so it cannot drift again.

**Architecture:** The bug is a producer/consumer contract mismatch in the per-staff layer of `synthetic_v2`. `_write_staff_crops` filters staff crops by ink-quality, then the per-staff manifest writer at `generate_synthetic.py:2400+` assigns `staff_index = enumerate(survivors)`, renumbering after drops. Plan A's systems builder (`yolo_aligned_systems.py:54`) computes expected indices as a cumsum over `staves_per_system` (physical position). The two disagree whenever a mid-page crop is filtered out, silently bundling wrong-staff tokens into surviving systems. The fix is to use the already-available `source_staff_index` (pre-filter physical position) when writing manifest entries, and to emit one entry per physical staff position regardless of whether the crop survived the filter. Plan A's stitching logic is unchanged. We add a contract test that fails if any future drift returns.

**Tech Stack:** Python 3.14, pytest, PowerShell remoting to GPU box (10.10.1.29), Verovio (existing dep).

**Already verified — do not redo:**
- `synthetic_systems_v1` has confirmed alignment corruption. Evidence: page `Abbott___Jane_Bingham_____Just_for_Today__lc6583477.mxl__bravura-compact__p001`, `sys01.png` image visually contains physical staves [3, 4, 5] but its token sequence describes [3, 5, 6]. Source: post-filter contiguous `staff_index` assignment in `generate_synthetic.py:2400-2437`, mismatched against Plan A's physical-position cumsum lookup in `yolo_aligned_systems.py:54`.
- Code-layer audit confirmed: `generate_synthetic.py` writes only per-staff artifacts (no system-level emission); Plan A reads only the per-staff manifest + labels_systems jsons. Files are physically separated; the leak is only via the silent index contract.
- `_write_staff_crops` already returns `List[Tuple[Path, physical_position]]` (line 2140: `(crop_path, max(0, source_index - 1))`), so the data we need is available; the writer just discards it.

```python
#!/usr/bin/env python3
"""Audit a per-staff manifest for alignment correctness.

Checks (no labels-root):
- For each page, are staff_index values contiguous starting at 0?
- Distribution of entries-per-page.

Checks (with labels-root):
- Does each page's manifest entry count match the label file's line count?
  Mismatch indicates post-filter renumbering hiding dropped staves.
"""
from __future__ import annotations

import argparse
import json
from collections import Counter, defaultdict
from pathlib import Path


def _load_manifest_indices(manifest_path: Path) -> dict[str, list[int]]:
    by_page: dict[str, list[int]] = defaultdict(list)
    with manifest_path.open("r", encoding="utf-8") as fh:
        for line in fh:
            line = line.strip()
            if not line:
                continue
            entry = json.loads(line)
            by_page[entry["page_id"]].append(int(entry["staff_index"]))
    return by_page


def _count_label_lines(labels_root: Path, page_id: str) -> int | None:
    candidates = list(labels_root.rglob(f"{page_id}.txt"))
    if not candidates:
        return None
    label_file = candidates[0]
    return sum(1 for line in label_file.read_text().splitlines() if line.strip())


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--manifest", type=Path, required=True)
    parser.add_argument("--labels-root", type=Path, default=None,
                        help="If given, compare per-page manifest entry count to label file line count.")
    parser.add_argument("--output", type=Path, required=True)
    parser.add_argument("--sample-size", type=int, default=20,
                        help="How many drifted pages to include in the report sample.")
    args = parser.parse_args()

    by_page = _load_manifest_indices(args.manifest)
    pages_total = len(by_page)
    pages_with_gap = 0
    pages_non_contiguous = 0
    sample_drift: list[dict] = []

    for page_id, indices in by_page.items():
        unique_sorted = sorted(set(indices))
        expected = list(range(len(unique_sorted)))
        if unique_sorted != expected:
            pages_non_contiguous += 1
            if max(unique_sorted) + 1 != len(unique_sorted):
                pages_with_gap += 1
            if len(sample_drift) < args.sample_size:
                sample_drift.append({"page_id": page_id, "staff_indices": unique_sorted})

    entries_per_page = Counter(len(v) for v in by_page.values())

    label_mismatch_pages = 0
    sample_label_mismatch: list[dict] = []
    if args.labels_root is not None:
        for page_id, indices in by_page.items():
            manifest_count = len(set(indices))
            label_count = _count_label_lines(args.labels_root, page_id)
            if label_count is not None and label_count != manifest_count:
                label_mismatch_pages += 1
                if len(sample_label_mismatch) < args.sample_size:
                    sample_label_mismatch.append({
                        "page_id": page_id,
                        "manifest_count": manifest_count,
                        "label_count": label_count,
                    })

    args.output.parent.mkdir(parents=True, exist_ok=True)
    args.output.write_text(json.dumps({
        "manifest": str(args.manifest),
        "labels_root": str(args.labels_root) if args.labels_root else None,
        "pages_total": pages_total,
        "pages_with_index_gap": pages_with_gap,
        "pages_with_non_contiguous_indices": pages_non_contiguous,
        "pages_with_label_count_mismatch": label_mismatch_pages,
        "entries_per_page": {str(k): v for k, v in sorted(entries_per_page.items())},
        "sample_drifted_pages": sample_drift,
        "sample_label_mismatch_pages": sample_label_mismatch,
    }, indent=2))
    print(f"[audit] manifest={args.manifest} pages={pages_total} "
          f"non_contiguous={pages_non_contiguous} label_mismatch={label_mismatch_pages}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
def test_process_page_systems_handles_non_contiguous_physical_indices(tmp_path: Path) -> None:
    """Contract: per-staff manifest staff_index is physical position (not post-filter enumerate).

    A page with 9 physical staves where physical positions 4 and 7 are missing
    from the manifest must:
    - Successfully assemble system 0 (staves 0,1,2: all present).
    - Drop system 1 (staves 3,4,5: position 4 missing) via token_miss.
    - Drop system 2 (staves 6,7,8: positions 7,8 missing) via token_miss.

    If the consumer code or producer regresses to post-filter contiguous semantics,
    this test fails because system 1 will silently use physical position 5's tokens
    in slot 4, producing wrong-staff content.
    """
    from src.data.yolo_aligned_systems import process_page_systems
```

```python
def _build_manifest_rows_for_page(
    *,
    page_basename: str,
    staff_crop_entries: List[Tuple["Path", int]],
    total_physical_staves: int,
    token_sequences_by_phys: Dict[int, List[str]],
    page_number: int,
    style_id: str,
    score_type: str,
    source_relpath: str,
    project_root: "Path",
    dataset_variants: Sequence[Tuple[str, str]],
) -> List[dict]:
    """Build per-staff manifest rows keyed by PHYSICAL staff position.

    For each physical staff position 0..total_physical_staves-1, emit a manifest
    entry. If the position has a surviving crop in `staff_crop_entries`, the
    `image_path` is set; otherwise `image_path` is None (the filter rejected the
    crop, but the token sequence is still derivable from the source split).

    Tokens come from `token_sequences_by_phys[physical_position]`. Callers are
    responsible for computing that lookup according to whichever code path
    (SVG primary or proportional fallback) they use.
    """
    crop_path_by_phys = {phys: path for path, phys in staff_crop_entries}

    rows: List[dict] = []
    for phys_idx in range(total_physical_staves):
        token_sequence = token_sequences_by_phys.get(phys_idx)
        if token_sequence is None:
            continue
        crop_path = crop_path_by_phys.get(phys_idx)
        image_path = relpath(project_root, crop_path) if crop_path is not None else None
        base_sample_id = f"{page_basename}__staff{phys_idx + 1:02d}"
        for dataset_name, sample_suffix in dataset_variants:
            sample_id = f"{base_sample_id}{sample_suffix}"
            rows.append(
                {
                    "sample_id": sample_id,
                    "dataset": dataset_name,
                    "split": "train",
                    "image_path": image_path,
                    "page_id": page_basename,
                    "source_path": source_relpath,
                    "style_id": style_id,
                    "page_number": page_number,
                    "staff_index": phys_idx,
                    "source_format": "musicxml",
                    "score_type": score_type,
                    "token_sequence": token_sequence,
                    "token_count": len(token_sequence),
                }
            )
    return rows
```
